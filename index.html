// Array of tasks for the role-play game const tasks = [ "Greet the other participant politely.", "Propose a meeting time.", "Introduce the agenda for the meeting.", "Politely reject a suggested time.", "Accept and confirm a meeting time.", "Discuss the duration of the meeting.", "Summarize and confirm the meeting details." ]; // Initialize the score let score = 0; // Ensure the page has loaded before accessing elements document.addEventListener("DOMContentLoaded", () => { // Select elements from the DOM const taskElement = document.getElementById("task"); const startButton = document.getElementById("startButton"); const rollDiceButton = document.getElementById("rollDice"); const diceRollElement = document.getElementById("diceRoll"); const scoreElement = document.getElementById("score"); // Start button functionality startButton.addEventListener("click", () => { // Pick a random task const randomTask = tasks[Math.floor(Math.random() * tasks.length)]; taskElement.innerText = `Task: ${randomTask}`; rollDiceButton.style.display = "inline-block"; // Show dice button diceRollElement.innerText = ""; // Clear previous dice roll }); // Roll dice functionality rollDiceButton.addEventListener("click", () => { // Generate a random dice roll between 1 and 6 const dice = Math.ceil(Math.random() * 6); diceRollElement.innerText = `You rolled: ${dice}`; // Update the score based on the dice roll if (dice > 3) { score += 10; alert("Great job! You scored 10 points!"); } else { alert("Keep trying! No points this time."); } // Update the score display scoreElement.innerText = `Score: ${score}`; }); });
